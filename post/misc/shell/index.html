zsh

bindkey

# 现有键映射名称的列表
bindkey -l
# 显示该键映射的全部映射
bindkey -M emacs
# 设置zsh的键映射
bindkey -e emacs
bindkey -e '^?' delete-char
# 清空所有键映射
bindkey -d

zle

## zle组件调用
zle backward-delete-char

git仓库关闭zsh插件读取git信息导致zsh卡顿

# 设置oh-my-zsh不读取文件变化信息
git config --add oh-my-zsh.hide-dirty 1
# 设置oh-my-zsh不读取任何git信息
git config --add oh-my-zsh.hide-status 1
